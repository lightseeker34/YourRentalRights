                    stars: "1.80 / 5",

                    issue: "Habitability & Mold",
                    url: "https://www.bbb.org/us/tx/austin/profile/property-management/main-street-renewal-llc-0825-1000101775"
                }
            };

            const selector = document.getElementById('landlord-select');
            const display = document.getElementById('stats-display');
            const linkEl = document.getElementById('bbb-link');
            const starEl = document.getElementById('customer-stars');
            const complaintsEl = document.getElementById('complaint-count');
            const issueEl = document.getElementById('common-issue');

            function updateDashboard(selectedKey) {
                const data = companyData[selectedKey];
                if (data) {
                    linkEl.innerText = data.letterRating;
                    starEl.innerText = data.stars;
                    complaintsEl.innerText = data.complaints;
                    issueEl.innerText = data.issue;
                    linkEl.href = data.url;

                    const starVal = parseFloat(data.stars);
                    if (starVal < 2.0) {
                        starEl.className = 'stat-value bad';
                    } else {
                        starEl.className = 'stat-value';
                    }
                    display.style.display = 'block';
                }
            }

            selector.addEventListener('change', function() {
                updateDashboard(this.value);
            });

            if (selector.value && selector.value !== "") {
                updateDashboard(selector.value);
            }
        });

        function startDocumentation() {
            const selector = document.getElementById('landlord-select');
            const company = selector.options[selector.selectedIndex].text;
            alert("Redirecting to documentation portal for: " + company);
        }
